# 1. 설치 (5분)
cd local-shorts-system
python -m venv venv
source venv/bin/activate
pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu118
pip install -r requirements.txt
python scripts/install_models.py  # 15.3 GB
ollama pull llama3.1:8b           # 4.7 GB

# 2. 실행
ollama serve                       # 터미널 1
cd backend && python app.py        # 터미널 2

# 3. 사용
브라우저: http://localhost:8000/docs
또는
curl -X POST http://localhost:8000/api/shorts/generate \
  -d '{"url": "https://example.com", "character_id": "executive-fox"}'
